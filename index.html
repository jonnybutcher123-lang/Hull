  <div id="map"></div>


  <script>

    const map = L.map('map').setView([53.75, -0.35], 12);


    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

      attribution: '&copy; OpenStreetMap contributors'

    }).addTo(map);


    // Colour scale (26 colours)

    const palette = [

      '#edf8fb', '#b2e2e2', '#66c2a4', '#2ca25f', '#006d2c',

      '#f7fcf5', '#e5f5e0', '#c7e9c0', '#a1d99b', '#74c476',

      '#31a354', '#006d2c', '#fff7ec', '#fee8c8', '#fdd49e',

      '#fdbb84', '#fc8d59', '#ef6548', '#d7301f', '#990000',

      '#d0d1e6', '#a6bddb', '#74a9cf', '#2b8cbe', '#045a8d', '#023858'

    ];


    const getColor = seg => palette[(seg - 1) % palette.length];


    fetch('Hull_OAs.geojson')

      .then(res => res.json())

      .then(data => {

        const layer = L.geoJSON(data, {

          style: feature => ({

            color: '#333',

            weight: 0.5,

            fillColor: getColor(feature.properties.segment || 1),

            fillOpacity: 0.8

          }),

          onEachFeature: (feature, layer) => {

            const seg = feature.properties.segment || '–';

            layer.bindPopup(`<b>OA:</b> ${feature.properties.OA21CD}<br><b>Segment:</b> ${seg}`);

          }

        }
